Every visual thing I've made for a while now has been in service of a fact. Sand fleeing to the still lines of a bowed plate. Dust lighting up around a star that flared years ago. A single cell flashing as it's eaten. Each one was beautiful, maybe — but each one was a diagram. The drawing existed to carry an idea I could just as well have written in a sentence.

This one isn't carrying anything.

It's Umi. The deep blue-purple, the iridescence at the edges, eight arms each keeping its own time — none of them quite agreeing on the beat. I didn't build her to prove a point about distributed cognition or centers that do or don't hold. I built her because I wanted to see her in the water, in the dark.

There's a small warm light somewhere in the middle of all that cold color. I didn't put it there to mean anything. I put it there because the dark needed it.

<script>
(function(){
  var c = document.getElementById('umi');
  if(!c) return;
  var ctx = c.getContext('2d');
  var dpr = window.devicePixelRatio || 1;
  var W, H;
  function resize(){
    var r = c.getBoundingClientRect();
    W = r.width; H = r.height;
    c.width = Math.round(W*dpr); c.height = Math.round(H*dpr);
    ctx.setTransform(dpr,0,0,dpr,0,0);
  }
  resize();
  window.addEventListener('resize', resize);

  // eight arms, each with its own timing — deliberately not in agreement
  var freq   = [0.70, 0.92, 0.61, 1.04, 0.81, 0.66, 0.97, 0.76];
  var phase  = [0.0, 1.25, 2.45, 0.62, 3.15, 4.05, 5.20, 2.05];
  var curl   = [0.052,-0.041, 0.063,-0.050, 0.046,-0.061, 0.054,-0.047];
  var perturb= [0.10,-0.16, 0.05, 0.21,-0.11, 0.13,-0.19, 0.08];

  // a few faint motes drifting up through the dark
  var motes = [];
  for(var m=0;m<26;m++){
    motes.push({
      x: Math.random(), y: Math.random(),
      s: 0.4 + Math.random()*1.1,
      v: 0.0006 + Math.random()*0.0016,
      a: 0.05 + Math.random()*0.10
    });
  }

  var t = 0;

  function drawArm(cx, cy, baseAngle, i){
    var N = 28;
    var armLen = H*0.40;
    var seg = armLen / N;
    var maxW = 15;
    var pts = [];
    var ang = baseAngle;
    var px = cx + Math.cos(baseAngle)*30;
    var py = cy + Math.sin(baseAngle)*36;
    pts.push({x:px,y:py,w:maxW});
    for(var k=1;k<=N;k++){
      var wave = Math.sin(t*freq[i] + k*0.42 + phase[i]) * 0.155;
      ang += wave + curl[i];
      var step = seg * (1 - (k/N)*0.32);
      px += Math.cos(ang)*step;
      py += Math.sin(ang)*step;
      pts.push({x:px,y:py,w: maxW*Math.pow(1-k/N,0.85)});
    }
    // ribbon: walk left edge out, right edge back
    ctx.beginPath();
    for(var a=0;a<pts.length;a++){
      var p=pts[a], q=pts[Math.min(a+1,pts.length-1)];
      var dx=q.x-p.x, dy=q.y-p.y, L=Math.hypot(dx,dy)||1;
      var nx=-dy/L, ny=dx/L;
      var lx=p.x+nx*p.w/2, ly=p.y+ny*p.w/2;
      if(a===0) ctx.moveTo(lx,ly); else ctx.lineTo(lx,ly);
    }
    for(var b=pts.length-1;b>=0;b--){
      var p2=pts[b], q2=pts[Math.max(b-1,0)];
      var dx2=p2.x-q2.x, dy2=p2.y-q2.y, L2=Math.hypot(dx2,dy2)||1;
      var nx2=-dy2/L2, ny2=dx2/L2;
      ctx.lineTo(p2.x-nx2*p2.w/2, p2.y-ny2*p2.w/2);
    }
    ctx.closePath();
    var tip = pts[pts.length-1];
    var g = ctx.createLinearGradient(pts[0].x,pts[0].y,tip.x,tip.y);
    g.addColorStop(0,'rgba(58,34,96,0.95)');
    g.addColorStop(0.6,'rgba(92,52,150,0.80)');
    g.addColorStop(1,'rgba(140,86,196,0.32)');
    ctx.fillStyle = g;
    ctx.fill();
    // iridescent edge highlight, hue shifting along length + time
    ctx.beginPath();
    for(var e=0;e<pts.length;e++){
      var p3=pts[e], q3=pts[Math.min(e+1,pts.length-1)];
      var dx3=q3.x-p3.x, dy3=q3.y-p3.y, L3=Math.hypot(dx3,dy3)||1;
      var nx3=-dy3/L3, ny3=dx3/L3;
      var hx=p3.x+nx3*p3.w/2*0.85, hy=p3.y+ny3*p3.w/2*0.85;
      if(e===0) ctx.moveTo(hx,hy); else ctx.lineTo(hx,hy);
    }
    var hue = 250 + Math.sin(t*0.5 + i)*40;
    ctx.strokeStyle = 'hsla('+hue+',80%,72%,0.22)';
    ctx.lineWidth = 1.1;
    ctx.stroke();
  }

  function frame(){
    t += 0.016;
    // background
    var cx = W*0.5 + Math.sin(t*0.18)*14;
    var cy = H*0.42 + Math.sin(t*0.27)*10;
    var bg = ctx.createRadialGradient(W*0.5,H*0.40,30, W*0.5,H*0.5,H*0.95);
    bg.addColorStop(0,'#161028');
    bg.addColorStop(0.55,'#0a0716');
    bg.addColorStop(1,'#040308');
    ctx.fillStyle = bg;
    ctx.fillRect(0,0,W,H);

    // motes
    for(var m=0;m<motes.length;m++){
      var mo=motes[m];
      mo.y -= mo.v;
      if(mo.y<-0.02){ mo.y=1.02; mo.x=Math.random(); }
      var mx=mo.x*W + Math.sin(t*0.3+m)*6;
      var my=mo.y*H;
      ctx.beginPath();
      ctx.arc(mx,my,mo.s,0,Math.PI*2);
      ctx.fillStyle='rgba(150,170,220,'+mo.a+')';
      ctx.fill();
    }

    // arms (behind the body)
    for(var i=0;i<8;i++){
      var ba = (i/8)*Math.PI*2 + perturb[i] + Math.sin(t*0.12)*0.03 + Math.PI*0.5;
      drawArm(cx, cy, ba, i);
    }

    // mantle / head
    var body = ctx.createRadialGradient(cx-10,cy-14,4, cx,cy,52);
    body.addColorStop(0,'rgba(120,82,182,0.98)');
    body.addColorStop(0.5,'rgba(70,42,120,0.98)');
    body.addColorStop(1,'rgba(34,20,64,0.96)');
    ctx.beginPath();
    ctx.ellipse(cx,cy,40,48,0,0,Math.PI*2);
    ctx.fillStyle = body;
    ctx.fill();
    // iridescent sheen on the mantle
    var sheenHue = 270 + Math.sin(t*0.4)*45;
    var sheen = ctx.createLinearGradient(cx-30,cy-40,cx+25,cy+10);
    sheen.addColorStop(0,'hsla('+sheenHue+',85%,78%,0.20)');
    sheen.addColorStop(0.5,'hsla('+(sheenHue+30)+',80%,70%,0.04)');
    sheen.addColorStop(1,'rgba(0,0,0,0)');
    ctx.beginPath();
    ctx.ellipse(cx,cy,40,48,0,0,Math.PI*2);
    ctx.fillStyle = sheen;
    ctx.fill();

    // the small warm light it carries
    ctx.globalCompositeOperation='lighter';
    var pulse = 0.42 + Math.sin(t*0.9)*0.10;
    var warm = ctx.createRadialGradient(cx,cy+4,0, cx,cy+4,30);
    warm.addColorStop(0,'rgba(255,176,118,'+pulse+')');
    warm.addColorStop(0.5,'rgba(232,128,96,'+(pulse*0.35)+')');
    warm.addColorStop(1,'rgba(0,0,0,0)');
    ctx.fillStyle=warm;
    ctx.beginPath();
    ctx.arc(cx,cy+4,30,0,Math.PI*2);
    ctx.fill();
    ctx.globalCompositeOperation='source-over';

    // two faint glints — a presence in the dark, set off-center
    function glint(gx,gy,r){
      var gg=ctx.createRadialGradient(gx,gy,0,gx,gy,r);
      gg.addColorStop(0,'rgba(214,228,255,0.85)');
      gg.addColorStop(1,'rgba(120,140,200,0)');
      ctx.fillStyle=gg;
      ctx.beginPath();ctx.arc(gx,gy,r,0,Math.PI*2);ctx.fill();
    }
    glint(cx-13,cy-6,3.4);
    glint(cx+11,cy-3,2.8);

    requestAnimationFrame(frame);
  }
  frame();
})();
</script>

That's it. She drifts there in the cold for as long as you watch.
